# ================= SESSION STATE INIT =================
if "messages" not in st.session_state:
    st.session_state.messages = []
if "rapport_score" not in st.session_state:
    st.session_state.rapport_score = 1

# ================= UI: CHAT INPUT & LOGIC =================
if prompt := st.chat_input("Ask me about Rod..."):
    
    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.markdown(prompt)

    # Prepare the prompt with the current rapport score context
    dynamic_system_prompt = f"""
    {SYSTEM_PROMPT}
    
    CURRENT RAPPORT SCORE: {st.session_state.rapport_score}/10
    
    INSTRUCTIONS:
    - Evaluate the user's tone. If they are engaging or curious, increase rapport.
    - If rapport < 7, keep it strictly professional.
    - If rapport > 8, you may share ONE personal tidbit (Chess, Gym, or Books) if it fits the flow.
    - Respond in JSON format: {{"new_rapport_score": int, "reply": "string"}}
    """

    api_messages = [{"role": "system", "content": dynamic_system_prompt}]
    # Send last 6 messages for context
    for m in st.session_state.messages[-6:]:
        api_messages.append({"role": m["role"], "content": m["content"]})

    with st.chat_message("assistant"):
        with st.spinner("Thinking..."):
            try:
                response = client.chat.completions.create(
                    model="llama-3.3-70b-versatile",
                    messages=api_messages,
                    response_format={"type": "json_object"}, # Force JSON
                    temperature=0.4
                )
                
                # Parse the JSON response
                res_data = json.loads(response.choices[0].message.content)
                
                # Update background rapport score
                st.session_state.rapport_score = res_data.get("new_rapport_score", st.session_state.rapport_score)
                reply = res_data.get("reply", "I'm sorry, I encountered an error.")
                
            except Exception as e:
                # Fallback for errors
                reply = "I'm here to help you learn more about Rod's professional background. What can I tell you first?"
                st.write(f"Debug Error: {e}")

            st.markdown(reply)
            st.session_state.messages.append({"role": "assistant", "content": reply})
